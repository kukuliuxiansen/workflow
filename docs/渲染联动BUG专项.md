### 测试代码

```javascript
// 测试切换工作流时的联动
async function testWorkflowSwitch() {
  // 1. 打开历史面板
  document.getElementById('historyPanel').classList.add('show');

  // 2. 展开右侧面板
  document.getElementById('rightPanel').classList.remove('collapsed');

  // 3. 打开一个弹窗
  document.getElementById('createModal').classList.add('show');

  // 4. 切换工作流
  await selectWorkflow('wf_xxx');

  // 5. 验证
  assert(!document.getElementById('historyPanel').classList.contains('show'), '历史面板应关闭');
  assert(document.getElementById('rightPanel').classList.contains('collapsed'), '右侧面板应收起');
  assert(!document.getElementById('createModal').classList.contains('show'), '弹窗应关闭');
}
```
